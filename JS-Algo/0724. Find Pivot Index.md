/**
 * @param {number[]} nums
 * @return {number}
 */
var pivotIndex = function(nums) {
    let leftSum = 0;
    let rightSum = 0;
    let ptr = 0;

    while(ptr < nums.length) {

        for(let i = 0; i < ptr; i++) {
            leftSum += nums[i];
        }

        for(let j = ptr + 1; j < nums.length; j++) {
            rightSum += nums[j];
        }

        if(leftSum === rightSum) {
            return ptr;
        } else {
            leftSum = 0;
            rightSum = 0;
        }

        ptr++;
    }
    return -1;
};
